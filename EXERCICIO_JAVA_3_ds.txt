package metodosCorretoPt2;

import java.util.Random;
public class InteirosSorteados {
	
	public int[] sortearNumeros() {
		Random r = new Random();

		int s1 = r.nextInt(100) + 1;
		int s2 = r.nextInt(100) + 1;

		return new int[]{s1, s2};
	}
	
	public boolean comparaNumeros(int n1, int n2) {

	    int[] sorteados = sortearNumeros();

	    int s1 = sorteados[0];
	    int s2 = sorteados[1];

	    System.out.println("Sorteados: " + s1 + " e " + s2);

	    if (n1 == s1 && n2 == s2) {
	        return true;
	    } else {
	        return false;
	    }
	}
}


//Esse método recebe dois números digitados pelo usuário (n1 e n2).
//Dentro dele, o programa sorteia dois novos números aleatórios (s1 e s2).
//Depois disso, ele compara:
//- se o primeiro número digitado é igual ao primeiro sorteado
//- e se o segundo número digitado é igual ao segundo sorteado
//Se os dois forem iguais ao mesmo tempo, retorna true.
//Caso contrário, retorna false.
//Ou seja: ele verifica se o usuário acertou exatamente os dois números sorteados.
